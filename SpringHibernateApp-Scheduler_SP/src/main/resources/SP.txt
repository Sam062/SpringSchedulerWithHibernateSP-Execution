CREATE DEFINER=`root`@`localhost` PROCEDURE `getAllEmployees`()
BEGIN
INSERT INTO employee_backup(emp_backup_id, emp_id, created_date, updated_date)
SELECT emp_id, emp_id, sysdate(), sysdate()  FROM employee
WHERE datediff(sysdate(), created_date)>= 2555;

DELETE FROM employee 
WHERE
    DATEDIFF(SYSDATE(), created_date) >= 2555;

END
